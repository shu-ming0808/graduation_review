```javascript
const uploadArea = document.getElementById('uploadArea');
const fileInput = document.getElementById('fileInput');
const studentInfo = document.getElementById('studentInfo');
const infoGrid = document.getElementById('infoGrid');
const cardsContainer = document.getElementById('cardsContainer');

uploadArea.addEventListener('click', () => fileInput.click());
uploadArea.addEventListener('dragover', (e) => {
  e.preventDefault();
  uploadArea.classList.add('dragover');
});
uploadArea.addEventListener('dragleave', () => {
  uploadArea.classList.remove('dragover');
});
uploadArea.addEventListener('drop', (e) => {
  e.preventDefault();
  uploadArea.classList.remove('dragover');
  const file = e.dataTransfer.files[0];
  if (file) handleFile(file);
});
fileInput.addEventListener('change', (e) => {
  const file = e.target.files[0];
  if (file) handleFile(file);
});

function handleFile(file) {
  const reader = new FileReader();
  reader.onload = (e) => {
    try {
      const data = JSON.parse(e.target.result);
      render(data);
    } catch (err) {
      alert('JSON 解析失敗：' + err.message);
    }
  };
  reader.readAsText(file, 'UTF-8');
}

function render(data) {
  const student = data[0]['課業學習'];
  renderStudentInfo(student.aboutMe, student.totalCredits);
  renderSemesterCards(student.gradeRecordList);
  uploadArea.classList.add('hidden');
}

function renderStudentInfo(about, totalCredits) {
  const items = [
    { label: '姓名', value: about.chineseName },
    { label: '英文姓名', value: about.englishName },
    { label: '學號', value: about.studentNumber },
    { label: '系級', value: about.departmentProgramGrade },
    { label: '主修', value: about.registerMajor },
    { label: '身份', value: about.studentIdentity },
    { label: '總修得學分', value: totalCredits },
  ];
  infoGrid.innerHTML = items.map(item =>
    `<div class="info-item"><span class="label">${item.label}：</span>${item.value}</div>`
  ).join('');
  studentInfo.style.display = 'block';
}

function renderSemesterCards(gradeRecordList) {
  cardsContainer.innerHTML = '';
  gradeRecordList.forEach(group => {
    const year = group.AcademicYear;
    const records = group.GradeRecords;

    const semesters = {};
    records.forEach(r => {
      const key = r.academicYear + '-' + r.semester;
      if (!semesters[key]) semesters[key] = [];
      semesters[key].push(r);
    });

    Object.keys(semesters).sort().reverse().forEach(key => {
      const [y, s] = key.split('-');
      const courses = semesters[key];
      const totalCredits = courses.reduce((sum, c) => sum + parseFloat(c.credit || 0), 0);

      const rows = courses.map(c => {
        const scoreText = getScoreDisplay(c.score);
        return `<tr>
          <td>${c.courseName}</td>
          <td style="text-align:center">${c.credit}</td>
          <td style="text-align:center" class="${scoreText.cls}">${scoreText.text}</td>
        </tr>`;
      }).join('');

      const card = document.createElement('div');
      card.className = 'semester-card';
      card.innerHTML = `
        <div class="card-header">${y} 學年度 第 ${s} 學期</div>
        <div class="card-body">
          <table>
            <thead><tr><th>課程名稱</th><th style="text-align:center">學分</th><th style="text-align:center">成績</th></tr></thead>
            <tbody>${rows}</tbody>
          </table>
        </div>
        <div class="card-footer">本學期合計 ${totalCredits} 學分</div>
      `;
      cardsContainer.appendChild(card);
    });
  });
}

function getScoreDisplay(score) {
  if (!score || score === '成績未到或無成績') return { text: '—', cls: 'score-pending' };
  if (score === '停修') return { text: '停修', cls: 'score-pending' };
  if (score === '通過') return { text: '通過', cls: 'score-pass' };
  if (score === '不通過') return { text: '不通過', cls: 'score-fail' };
  const num = parseFloat(score);
  if (isNaN(num)) return { text: score, cls: 'score-pending' };
  if (num < 60) return { text: num.toFixed(2), cls: 'score-fail' };
  return { text: num.toFixed(2), cls: 'score-pass' };
}
```
